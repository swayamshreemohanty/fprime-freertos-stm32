# Create a static library for STM32 HAL + FreeRTOS + Application code
add_library(stm32h7_hal STATIC 
    ${STM32_Drivers_Src}
    ${FreeRTOS_Src}
    ${MX_Application_Src}
)

# Set include directories
target_include_directories(stm32h7_hal PUBLIC ${MX_Include_Dirs})

# Set compile definitions
target_compile_definitions(stm32h7_hal PUBLIC ${MX_Defines_Syms})

# Set C standard for HAL code
set_target_properties(stm32h7_hal PROPERTIES
    C_STANDARD 11
    C_STANDARD_REQUIRED ON
    C_EXTENSIONS ON
)
